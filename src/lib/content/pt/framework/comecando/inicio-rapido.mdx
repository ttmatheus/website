## Passo 2: Criar o Controller ğŸ¯

Crie `src/controllers/UserController.ts`. Os controllers organizam suas rotas em classes lÃ³gicas:

```typescript
import { Controller, Get, Post, Body, Param, Res } from "azurajs/decorators";
import type { ResponseServer } from "azurajs/types";

const users = [
  { id: 1, name: "Alice" },
  { id: 2, name: "Bob" },
];

@Controller("/api/users")
export class UserController {
  @Get()
  getAllUsers(@Res() res: ResponseServer) {
    res.json({ success: true, data: users });
  }

  @Get("/:id")
  getUserById(@Param("id") id: string, @Res() res: ResponseServer) {
    const user = users.find(u => u.id === Number(id));
    if (!user) return res.status(404).json({ success: false, error: "Not found" });
    res.json({ success: true, data: user });
  }

  @Post()
  createUser(@Body() body: any, @Res() res: ResponseServer) {
    const newUser = { id: users.length + 1, ...body };
    users.push(newUser);
    res.status(201).json({ success: true, data: newUser });
  }
}
```
